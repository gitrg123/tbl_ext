text = """header : this is header1
response : this is response1

header : this is header2
response : this is response2
this paragraph is part of the same response2

header : this is header3
response : this is response3"""

# Split the text into lines
lines = text.split('\n')

# Initialize an empty dictionary
header_response_dict = {}

# Iterate through the lines
current_header = None
current_response = ""

for line in lines:
    # Split each line into parts using " : " as the separator
    parts = line.split(' : ')
    if len(parts) == 2:
        key, value = parts
        if key.strip() == 'header':
            # Set the current header
            current_header = value.strip()
        elif key.strip() == 'response':
            # Append to the current response
            current_response += value.strip() + ' '
    else:
        # If the line doesn't contain " : ", it's considered part of the current response
        current_response += line.strip() + ' '

    # If a new header is encountered, store the previous header and response in the dictionary
    if current_header and current_response:
        header_response_dict[current_header] = current_response.strip()
        current_header = None
        current_response = ""

# Print the resulting dictionary
print(header_response_dict)
