from docx import Document
from docx.shared import Pt
from docx.enum.text import WD_ALIGN_PARAGRAPH

# Create a new Word document
doc = Document()

# Add a paragraph with text
paragraph = doc.add_paragraph("Click here to visit OpenAI's website:")

# Set the paragraph alignment to center
paragraph.alignment = WD_ALIGN_PARAGRAPH.CENTER

# Add a hyperlink to the paragraph
hyperlink = paragraph.add_hyperlink("https://www.openai.com", "OpenAI's Website")

# Set the font size and style for the hyperlink
hyperlink.text_run.font.size = Pt(12)
hyperlink.text_run.font.italic = True
hyperlink.text_run.underline = True

# Save the document
doc.save("document_with_hyperlink.docx")
