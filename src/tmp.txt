from docx import Document
from docx.enum.text import WD_ALIGN_PARAGRAPH
from docx.shared import Pt
from docx.oxml.ns import nsdecls
from docx.oxml import parse_xml

# Create a new Word document
doc = Document()

# Add a paragraph with text
paragraph = doc.add_paragraph("Click here to visit OpenAI's website:")

# Set the paragraph alignment to center
paragraph.alignment = WD_ALIGN_PARAGRAPH.CENTER

# Create a run for the hyperlink text
run = paragraph.runs[0]

# Create the hyperlink
hyperlink = run._r
hyperlink.get_or_add_hyperlink("https://www.openai.com", rId=1, nsdecls={'r': nsdecls['r']})

# Set the font size, style, and color for the hyperlink
run.font.size = Pt(12)
run.font.italic = True
run.font.underline = True
run.font.color.rgb = parse_xml('FF0000')  # Red color

# Save the document
doc.save("document_with_hyperlink.docx")
